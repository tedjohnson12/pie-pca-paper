% Abstract with filler text
\begin{abstract}
    The Planetary Infrared Excess method to measure the thermal emission of an exoplanet is analogous to
    detecting the infrared excess of a circumstellar disk or low-mass companion -- except that the 
    exceptionally low contrast between planet and star require extreme precision and make the method
    easily susceptible to bias. We propose a new method to analyze PIE time-series data that naturally
    accounts for stellar variability and is agnostic to stellar spectral models. This method treats
    the observation as a vector space and isolates variability present at long wavelengths that is
    uncorrelated to short-wavelength variability.
\end{abstract}

\subsection{Motivation}
We take a vector-based approach to time-series spectroscopy, representing spectra as vectors and observations as matrices.
\subsubsection{Spectroscopic Bases}
\label{subsubsec:bases}
A spectrograph discretizes a continuous spectrum into a series of wavelength bins. These spectral
elements are approximately linearly independent (though their values may be highly correlated). We can therefore
discuss a 1D spectrum as a vector of $n$ dimensions, where $n$ is the number of spectral elements. Let $S$ be a vector space of $n$ dimensions. A spectrum $\bm{f}$ can be written as
\begin{equation}
    \bm{f} = [f_1,\,f_2,\,...,\,f_j,\,...,\,f_n]
\end{equation}
where $f_j$ is a scalar value representing the flux of a single spectral channel.

Suppose the surface of a rotating star is made up of two regions:
\begin{enumerate}
    \item A quiet photosphere with intrinsic spectrum $\bm{f}_p$
    \item Low temperature spots, with intrinsic spectrum $\bm{f}_s$
\end{enumerate}

The fraction of each region visible to the observer is dependent on their viewing angle, and will change as the star rotates leading to a time-variable stellar spectrum. It can be written
\begin{equation}
    \label{eq:spec_basis_single}
    \bm{f}(t) = a_p(t) \bm{f}_p + a_s(t) \bm{f}_s
\end{equation}
where $a_p$ and $a_s$ are scalar functions of time. If this star is observed at times $\bm{t} = [t_1,\,t_2,\,t_3]$, then the resulting spectra will be 
\begin{equation}
    \label{eq:spec_basis_triple}
    \bm{f}(\bm{t})
    = \begin{bmatrix}
        a_p(t_1) \bm{f}_p + a_s(t_1) \bm{f}_s \\
        a_p(t_2) \bm{f}_p + a_s(t_2) \bm{f}_s \\
        a_p(t_3) \bm{f}_p + a_s(t_3) \bm{f}_s
    \end{bmatrix}
\end{equation}

Note that the right-hand sides of Equations (\ref{eq:spec_basis_single} \& \ref{eq:spec_basis_triple}) could be more concisely written if we define a basis $B = \{\bm{f}_p,\,\bm{f}_s\}$. In this basis, we can rewrite Equation (\ref{eq:spec_basis_single})
\begin{equation}
    [\bm{f}(t)]_{B} 
    = \begin{bmatrix}
        a_p(t),\,a_s(t)
    \end{bmatrix}
\end{equation}
And Equation \ref{eq:spec_basis_triple} is written
\begin{equation}
    [\bm{f}(\bm{t})]_{B} 
    = \begin{bmatrix}
        a_p(t_1) & a_s(t_1) \\
        a_p(t_2) & a_s(t_2) \\
        a_p(t_3) & a_s(t_3)
    \end{bmatrix}
\end{equation}


We generalize this to any astrophysical scene composed of objects whose respective scaling factors vary with time. For any time-variable scene whose variability can be written
\begin{equation}
    \bm{f}(t) = \sum_j a_j(t) \bm{f}_j
\end{equation}
a time series observation of $\bm{t} = [t_1,\,t_2,\,...\, t_i,\, ...,\,t_m]$ is then
\begin{equation}
    \label{eq:gen_spec_base_ts}
    \bm{f}(\bm{t})
    = \begin{bmatrix}
        \sum_j a_j(t_1) \bm{f}_j \\
        \sum_j a_j(t_2) \bm{f}_j \\
        ... \\
        \sum_j a_j(t_i) \bm{f}_j \\
        \dots \\
        \sum_j a_j(t_m) \bm{f}_j \\
    \end{bmatrix}
    = \mathcal{F}_{ij}
\end{equation}
where $\mathcal{F}_{ij}$ is a matrix of shape $m\times n$ whose rows are spectra and whose columns are light curves for a single spectral channel. We then define a basis $B = \{ \bm{f}_1,\,\bm{f}_2,\, ...,\, \bm{f}_j,\, ...,\, \bm{f}_p \}$ and use it to rewrite Equation (\ref{eq:gen_spec_base_ts})
\begin{equation}
    [\bm{f}(\bm{t})]_B 
    = \begin{bmatrix}
        a_1(t_1) & a_2(t_1) & ... & a_j(t_1) & ... & a_p(t_1) \\
        a_1(t_2) & a_2(t_2) & ... & a_j(t_2) & ... & a_p(t_2) \\
        \dots    & \dots    & ... & \dots    & ... & \dots    \\
        a_1(t_i) & a_2(t_i) & ... & a_j(t_i) & ... & a_p(t_i) \\
        \dots    & \dots    & ... & \dots    & ... & \dots    \\
        a_1(t_m) & a_2(t_m) & ... & a_j(t_m) & ... & a_p(t_m) \\
    \end{bmatrix}
\end{equation}
or, more concisely
\begin{equation}
    [\mathcal{F}_{ij}]_B = a_j(t_i)
\end{equation}

Where $[\mathcal{F}_{ij}]_B$ is an $m\times p$ matrix, where $p$ is the number of independent spectra that make up this variable scene. Herein we will assume that $p$ is small compared to $m$, that is, there are many more observed epochs than there are independent components in a model.

In practice, the basis $B$ is unknown. A spectroscopic time series resulting from an observation only contains scene-integrated spectra, and the actual spectra of the components ($\bm{f}_j$) are unknown. The time-series is written
\begin{equation}
    \label{eq:gen_obs_ts}
    \bm{f}(\bm{t})
    = \begin{bmatrix}
        \bm{f}(t_1) \\
        \bm{f}(t_2) \\
        ... \\
        \bm{f}(t_i) \\
        ... \\
        \bm{f}(t_m) \\
    \end{bmatrix}
\end{equation}
However, there is some set of $q \mid q \le p$ vectors $\bm{f}(t_i)$ that span this vector space. We define a set of indices $s$ to describe the epochs in which these vectors occur. We can then write the basis $B = \{\bm{f}(t_i)~\forall~i~\mid~i~\, \in\, s\}$ and use it to rewrite Equation (\ref{eq:gen_obs_ts}).
\begin{equation}
    [\bm(f)(\bm{t})]_B
    = \begin{bmatrix}
        b_i(\bm{t})\quad\forall ~i~\mid~i\,\in\,s
    \end{bmatrix}
\end{equation}
where $b_i(\bm{t})$ is the time-dependent coefficient on each of the basis vectors. The observation, initially described by an $m\times n$ matrix in Equation \ref{eq:gen_spec_base_ts}, is now described by two matrices: 1) a $q\times n$ matrix describing the basis, and 2) a $m\times q$ matrix describing the time-series in that basis. Any of our original vectors $\bm{f}(t_i)$ can be represented by a sum of the basis vectors
\begin{equation}
    \label{eq:single_epoch_sum}
    \bm{f}(t_i) = \sum_{i' \in s} b_{i'}(t_i)\bm{f}(t_{i'}) + \bm{\epsilon}(t_i)
\end{equation}
where $\bm{\epsilon}$ is an error term that in practice arises due to noise and deviations of an actual astrophysical scene from our strict assumptions. Section \ref{subsec:implementation} will discuss methods to choose the set $s$ and the coefficients $b_{i'}(t_i)$ such that $\bm{\epsilon}^2$ is minimized.

\subsubsection{Application to PIE}

A prototypical PIE observation consists of an unresolved astrophysical scene consisting of a star and a planet observed across two bands, the near
infrared (NIR) and mid infrared (MIR). The star -- being much hotter than the planet -- contributes flux to both bands. The planet is not hot enough
to contribute to the NIR band, and only produces measurable flux in the MIR. We can write the flux in epoch $i$
\begin{equation}
    \label{eq:obs_composition}
    \bm{f}(t_i) = \bm{f}_s(t_i) + \bm{f}_p(t_i)
\end{equation}
for time-dependent stellar spectrum $\bm{f}_s(t_i)$ and time-dependent planetary spectrum $\bm{f}_p(t_i)$. In the NIR (i.e. where the planet contributes 0 flux), this is written
\begin{equation}
    \bm{f}^{\text{NIR}}(t_i) = \bm{f}^\text{NIR}_s(t_i)
\end{equation}

Considering this star-only portion of the spectrum, we can write it similar to Equation (\ref{eq:single_epoch_sum}).
\begin{equation}
    \begin{array}{ccl}
        \bm{f}^{\text{NIR}}(t_i) &=& \sum_{i' \in s} b_{i'}(t_i) \bm{f}^\text{NIR}(t_i') + \bm{\epsilon}^\text{NIR}(t_i) \\
        &=& \tilde{\bm{f}}^{\text{NIR}}(t_i) + \bm{\epsilon}^\text{NIR}(t_i)
    \end{array}
\end{equation}
where $s$ and $b_{i'}(t_i)$ are calculated using the NIR data only, and $\tilde{\bm{f}}$ represents the reconstruction of $\bm{f}$ using the basis $B$. We then use these NIR-only bases and coefficients to compute a reconstruction of the whole spectrum.
\begin{equation}
    \begin{array}{ccl}
        \bm{f}(t_i) &=& \sum_{i' \in s} b_{i'}(t_i) \bm{f}(t_i') + \bm{\epsilon}(t_i) + \bm{\delta}(t_i) \\
        &=& \tilde{\bm{f}}(t_i) + \bm{\epsilon}(t_i) + \bm{\delta}(t_i)
    \end{array}
\end{equation}
Where $\bm{\delta}$ is an additional error term due to the uncorrelated variability of the planet. By subtracting $\tilde{\bm{f}}(t_i)$ from $\bm{f}(t_i)$ we are able to measure the quantity $\bm{\epsilon}(t_i) + \bm{\delta}(t_i)$. We can interpret this quantity by substituting in Equation (\ref{eq:obs_composition})
\begin{equation}
    \begin{array}{ccl}
        \bm{f}_s(t_i) + \bm{f}_p(t_i) &=& \sum_{i' \in s} b_{i'}(t_i) \bm{f}_s(t_i')  \\
        & & + \sum_{i' \in s} b_{i'}(t_i) \bm{f}_p(t_i') \\
        & & + \bm{\epsilon}(t_i) + \bm{\delta}(t_i)
    \end{array}
\end{equation}
or
\begin{equation}
    \bm{\epsilon}(t_i) + \bm{\delta}(t_i) = \bm{f}_s(t_i) - \tilde{\bm{f}_s}(t_i) + \bm{f}_p(t_i) - \tilde{\bm{f}_p}(t_i)
\end{equation}
Since $\bm{\epsilon}(t_i)$ is defined as $\bm{f}_s(t_i) - \tilde{\bm{f}_s}(t_i)$, we are left with a definition for $\bm{\delta}(t_i)$:
\begin{equation}
    \begin{array}{ccl}
        \bm{\delta}(t_i) &=& \bm{f}_p(t_i) - \tilde{\bm{f}_p}(t_i) \\
        &=& \bm{f}_p(t_i) - \sum_{i'\in s} b_{i'}(t_i) \bm{f}_p(t_i')
    \end{array}
\end{equation}

The quantity $\bm{\epsilon}(t_i) + \bm{\delta}(t_i)$ can be measured, however, only $\bm{\delta}(t_i)$ provides any information about the planet's spectrum. Therefore, $\bm{\epsilon}(t_i)$ is not important so long as it is small. For simplicity, we will assume that $\bm{\delta}(t_i) \gg \bm{\epsilon}(t_i)$ in the MIR, the only wavelength band in which $\bm{\delta}$ is defined. This assumption allows for a direct measurement of $\bm{f}_p(t_i) - \tilde{\bm{f}_p}(t_i)$. To compute $\tilde{\bm{f}_p}(t_i)$ from some planetary phase curve model $\bm{f}_p(t)$ requires the set of basis indices $s$ as well as the coefficients $b_{i}(t)$, both of which are computed using the NIR data only and are independent of the planetary model. Therefore, any forward model of a planetary phase curve can be used to predict the observable error term $\bm{\delta}(t)$. Section \ref{subsec:implementation} will use this to compare the measured $\bm{\delta}(t)$ to the expected $\bm{\delta}(t)$ given by a grid of planetary models in order to retrieve the characteristics of the observed planet.
